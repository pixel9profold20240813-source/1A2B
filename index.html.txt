    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.0.2/firebase-app.js";
        import { getFirestore, doc, setDoc, getDoc, updateDoc, onSnapshot, arrayUnion } from "https://www.gstatic.com/firebasejs/11.0.2/firebase-firestore.js";
        import { getAuth, signInAnonymously, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.0.2/firebase-auth.js";

        // 你的 Firebase 配置
        const firebaseConfig = {
            apiKey: "AIzaSyCmpCcJ3hQ5UeAzYirbwmTkCdrxwKpcbhU",
            authDomain: "a2b-dea5c.firebaseapp.com",
            projectId: "a2b-dea5c",
            storageBucket: "a2b-dea5c.firebasestorage.app",
            messagingSenderId: "719535143959",
            appId: "1:719535143959:web:41e767d89b8b75450af052"
        };

        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);
        const auth = getAuth(app);
        const collectionPath = 'artifacts/1a2b-game/public/data/rooms';

        let userId, currentRoomId, playerRole, gameData, unsubscribe = null;

        onAuthStateChanged(auth, u => {
            if (u) {
                userId = u.uid;
                document.getElementById('connection-status').innerText = "伺服器已連線";
            }
        });
        signInAnonymously(auth);

        const showScreen = id => {
            ['screen-lobby', 'screen-waiting', 'screen-game', 'screen-result'].forEach(s => document.getElementById(s).classList.add('hidden'));
            document.getElementById(id).classList.remove('hidden');
        };

        window.createRoom = async () => {
            const rid = Math.random().toString(36).substring(2, 8).toUpperCase();
            currentRoomId = rid; playerRole = 'p1';
            await setDoc(doc(db, collectionPath, rid), {
                p1: userId, p2: null, p1Secret: null, p2Secret: null, turn: 'p1', history: [], winner: null
            });
            startListening(rid);
            document.getElementById('display-room-id').innerText = rid;
            showScreen('screen-waiting');
        };

        window.joinRoom = async () => {
            const rid = document.getElementById('join-room-id').value.toUpperCase();
            if (rid.length !== 6) return alert("請輸入 6 位代碼");
            const snap = await getDoc(doc(db, collectionPath, rid));
            if (!snap.exists()) return alert("找不到房間");
            if (snap.data().p2 && snap.data().p2 !== userId) return alert("房間已滿");
            
            currentRoomId = rid;
            playerRole = (snap.data().p1 === userId) ? 'p1' : 'p2';
            if (!snap.data().p2 && snap.data().p1 !== userId) {
                await updateDoc(doc(db, collectionPath, rid), { p2: userId });
            }
            startListening(rid);
            document.getElementById('display-room-id').innerText = rid;
            showScreen('screen-waiting');
        };

        const startListening = rid => {
            if (unsubscribe) unsubscribe();
            unsubscribe = onSnapshot(doc(db, collectionPath, rid), snap => {
                if (!snap.exists()) return;
                gameData = snap.data();
                updateUI();
            });
        };

        const updateUI = () => {
            const d = gameData;
            // 等待室狀態
            if (d.p1) document.getElementById('p1-status').children[0].classList.replace('bg-gray-700', 'bg-cyan-500');
            if (d.p2) {
                document.getElementById('p2-status').children[0].classList.replace('bg-gray-700', 'bg-purple-500');
                showScreen('screen-game');
            }

            document.getElementById('player-role-badge').innerText = playerRole === 'p1' ? "PLAYER 1" : "PLAYER 2";
            
            const mySecret = playerRole === 'p1' ? d.p1Secret : d.p2Secret;
            if (!mySecret) {
                document.getElementById('setup-section').classList.remove('hidden');
                document.getElementById('play-section').classList.add('hidden');
                document.getElementById('turn-indicator').innerText = "請設定秘密數字";
            } else {
                document.getElementById('setup-section').classList.add('hidden');
                document.getElementById('play-section').classList.remove('hidden');
                if (!d.p1Secret || !d.p2Secret) {
                    document.getElementById('turn-indicator').innerText = "等待對方設定...";
                    document.getElementById('guess-btn').disabled = true;
                } else {
                    const isMyTurn = d.turn === playerRole;
                    document.getElementById('turn-indicator').innerText = isMyTurn ? "你的回合！" : "對方回合...";
                    document.getElementById('guess-btn').disabled = !isMyTurn;
                }
            }

            renderHistory(d.history);
            if (d.winner) {
                showScreen('screen-result');
                document.getElementById('result-title').innerText = d.winner === playerRole ? "VICTORY" : "DEFEAT";
                document.getElementById('result-msg').innerText = d.winner === playerRole ? "你贏了！" : "對方先猜到了...";
            }
        };

        const renderHistory = h => {
            const list = document.getElementById('history-list');
            list.innerHTML = h.map(i => `
                <div class="flex justify-between p-2 border-b border-white/5">
                    <span>${i.player === playerRole ? '你' : '對方'}: <b class="font-orbitron">${i.guess}</b></span>
                    <span class="font-bold text-cyan-400">${i.a}A ${i.b}B</span>
                </div>
            `).reverse().join('');
        };

        window.setSecret = async () => {
            const val = document.getElementById('secret-input').value;
            if (!/^\d{4}$/.test(val) || new Set(val).size !== 4) return alert("請輸入 4 位不重複數字");
            await updateDoc(doc(db, collectionPath, currentRoomId), playerRole === 'p1' ? { p1Secret: val } : { p2Secret: val });
        };

        window.submitGuess = async () => {
            const guess = document.getElementById('guess-input').value;
            if (!/^\d{4}$/.test(guess) || new Set(guess).size !== 4) return alert("請輸入 4 位不重複數字");
            
            const target = playerRole === 'p1' ? gameData.p2Secret : gameData.p1Secret;
            let a = 0, b = 0;
            guess.split('').forEach((g, i) => {
                if (g === target[i]) a++;
                else if (target.includes(g)) b++;
            });

            const update = {
                history: arrayUnion({ player: playerRole, guess, a, b }),
                turn: playerRole === 'p1' ? 'p2' : 'p1'
            };
            if (a === 4) update.winner = playerRole;
            await updateDoc(doc(db, collectionPath, currentRoomId), update);
            document.getElementById('guess-input').value = '';
        };
    </script>
